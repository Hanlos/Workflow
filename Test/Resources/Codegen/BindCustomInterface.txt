module test;
using test::*;
using system::*;

var s = "";

interface IAdder
{
	func Add(value : int) : void;
	func GetSum() : int;
	event SumChanged();
	prop Sum : int {GetSum : SumChanged}
}

func Execute(adder : IAdder^) : void
{
	var subscription = bind(adder.Sum);
	var listener = subscription.Subscribe(
		func (value : object) : void
		{
			s = $"$(s)[$(cast int value)]";
		}
	);

	for(x in range [1, 5])
	{
		adder.Add(x);
	}
}

func main() : string
{
	return s;
}
